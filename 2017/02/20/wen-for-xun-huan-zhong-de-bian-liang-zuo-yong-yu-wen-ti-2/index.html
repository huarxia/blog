问 for 循环中的变量作用域问题

for (var i = 0; i < 5; i++) {  
    setTimeout(function () {
        console.log('i=' + i);
    }, 0);
}

请问：输出结果是什么？

五个i=5

那么要实际输出1~5应该怎么修改？

for (let i = 0; i<=5; i++) {  
     setTimeout( function timer(){
         console.log('i=' + i);
     }, 0);
}

或者：

for (var i = 0; i<= 5; i++) {  
     (function(j){
         setTimeout( function() {
             console.log('j=' + j);
         }, 0);
     })(i);
}
