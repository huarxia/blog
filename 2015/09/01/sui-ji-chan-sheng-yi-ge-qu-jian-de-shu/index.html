随机产生一个区间的数

/*
  * 说明:取任意值开始 至 任意值的随机整数
  * 参数:
      under:起始值
      over: 终止值
*/
function fRandomBy(under, over){  
    switch(arguments.length){
        case 1: return parseInt(Math.random()*under+1);
        case 2: return parseInt(Math.random()*(over-under+1) + under); 
        default: return 0;
    }
}
